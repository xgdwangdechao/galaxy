187  bool Application::run() noexcept
188  {
189  // This is to ensure gameloop is running at 60 UPS, independant of FPS. 1.0 / 60.0.
190  std_chrono_tp current;
191  std_chrono_duration elapsed;
192  std::chrono::nanoseconds lag(0ns);
193  constexpr std::chrono::nanoseconds update_ratio(16ms);
194 
195  // Fixed timestep gameloop. Pretty easy to understand.
196  // Simply loop the game until the window closes, then the mainloop can handle restarting the application if restart = true.
197  auto previous = std_chrono_hrc::now();
198  while (m_window->is_open())
199  {
200  current = std_chrono_hrc::now();
201  elapsed = current - previous;
202  previous = current;
203  lag += elapsed;
204 
205  m_window->poll_events();
206  while (lag >= update_ratio)
207  {
208  m_delta_time.set(static_cast<double>(lag.count()) / static_cast<double>(update_ratio.count()));
209 
210  m_state->events();
211  m_state->update(m_delta_time.get());
212 
213  lag -= update_ratio;
214  }
215 
216  // Begin render.
217  m_window->begin();
218 
219  m_state->render();
220 
221  m_window->end();
222  // End render.
223  }
224 
225  m_state->clear();
226  m_threadpool->end();
227 
228  FTLIB.close();
229  m_window->destroy();
230 
231  PL_LOG_FINISH;
232 
233  return SL_HANDLE.m_restart;
234  }
